1. Identify which salesperson has generated the highest total sales.
>>>select sname from salespeople as s,customer as c,orders as o where s.snum=c.snum and c.cnum=o.cnum and o.amt=(select max(amt) from orders);


2. Count how many customers are handled by each salesperson.
>>>select count(*),s.sname from salespeople as s,customer as c where s.snum=c.snum group by s.sname;


3. Find out which customers have placed the most orders.
>>>select c.cname,count(o.onum) as order_count from customer as c,orders as o where c.cnum=o.cnum group by c.cnum order by order_count desc limit 3;


4. Analyze the average size of orders for each salesperson's customers.
>>>select sname, avg(amt) as avg_order_size from salespeople as s, orders as o, customer as c where s.snum = c.snum and c.cnum = o.cnum group by s.snum;


5. Identify the customers who have spent the most money on orders.
>>>select c.cname from customer as c,orders as o where c.cnum=o.cnum and o.amt=(select max(amt) from orders);


6. Analyze which cities (regions) generate the highest total sales.
>>>select s.city from salespeople as s, orders as o, customer as c where s.snum = c.snum and c.cnum = o.cnum and o.amt=(select max(amt) from orders);


7. Find out which salesperson has the highest average customer rating.
>>>select sname, avg(rating) as avg_rating from salespeople as s, customer as c where s.snum = c.snum group by s.snum order by avg_rating desc limit 1;


8. Calculate the percentage of customers who have placed orders for each salesperson.
>>>select sname,(select count(*) from orders where cnum = customer.cnum) / (select count(*) from customer where snum = salespeople.snum) * 100 as order_percentage from salespeople, customer where salespeople.snum = customer.snum;


9. Calculate the total revenue generated by each customer.
>>>select cname, (select sum(amt) from orders where cnum = customer.cnum) as total_revenue from customer;


10. Identify customers who have not placed any orders.
>>>select cname from customer where cnum not in (select distinct cnum from orders);


11. Show the percentage of total revenue generated by each salesperson.
>>>select sname,(select sum(o.amt) from orders as o, customer as c where o.cnum = c.cnum and c.snum = salespeople.snum) / (select sum(amt) from orders) * 100 as revenue_percentage from salespeople;
